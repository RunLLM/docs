[docs]class TableArtifact(Artifact):

    """This class represents a computed table within the flow's DAG.

    An artifact can only be created from a Flow object, or by applying
    operators to an existing artifact. For example:

        client = spiral.SpiralClient(...)
        flow = client.create_flow("foo")
        tableArtifactFromFlow = flow.sql("SELECT * ...")
        tableArtifactFromOperator = transform('function.zip', tableArtifactFromFlow)

        Alternatively:

        @op
        def foo(df):
            # the contents of in `function.zip` here
            return result

        tableArtifactFromOperator = foo(tableArtifactFromFlow)

    The contents of these artifacts can be manifested locally or written to an
    integration. For example:

        data = tableArtifact.to_df()
        tableArtifact.save("integration/output_table")

    The contents of these artifacts can also be tested. For example:

        passed, logs = table_artifact.add_unittest(
            name="foo",
            description="This is an illustrative example",
            file_path="unittest.zip",
        )

        Alternatively:

        @add_unittest
        def foo_test(df: pd.DataFrame):
            return df["year"] < 2022

        passed, logs = foo_test(table_artifact)
    """

    def __init__(
        self,
        api_client: APIClient,
        dag: DAG,
        artifact_id: uuid.UUID,
    ):
        self._api_client = api_client
        self._dag = dag
        self._artifact_id = artifact_id

        # Record the statuses of validations that have already run.
        self._validation_statuses: Dict[str, bool] = {}


[docs]    def to_df(self) -> pd.DataFrame:
        """Materializes a TableArtifact into an actual dataframe.

        Args:
            sample:
                The number of rows to restrict the output dataframe.

        Returns:
            A dataframe containing the tabular contents of this artifact.

        Raises:
            InvalidRequestError:
                An error occurred because of an issue with the user's code
                or sql queries, or the `sample` parameter is malformed.
            InternalServerError:
                An unexpected error occurred within the Spiral cluster.
        """
        dag = apply_deltas_to_dag(
            self._dag,
            deltas=[
                SubgraphDAGDelta(
                    artifact_ids=[self._artifact_id],
                    include_validation_operators=False,
                    include_load_operators=False,
                )
            ],
            make_copy=True,
        )

        preview_resp = self._api_client.preview(dag=dag)
        artifact_result = preview_resp.artifact_results[self._artifact_id]

        if artifact_result.table:
            # Translate the previewed table in a dataframe.
            return pd.DataFrame(json.loads(artifact_result.table.data)["data"])
        else:
            raise SpiralError("Artifact does not have table.")



[docs]    def save(self, config: SaveConfig) -> None:
        """
        Register an integration to write the contents of this artifact to.

        db_integration = spiral_client.load_integration(name="spiral_demo/")
        churn_data = db_integration.sql(...)
        churn_data.save(config=db_integration.config(table="test"))

        Args:
            config:
                SaveConfig object generated from integration using
                the <integration>.config(...) method.
        Raises:
            InvalidIntegrationException:
                An error occurred because the requested integration could not be
                found.
        """
        integration_info = config.integration_info
        integration_load_params = config.parameters
        integrations_map = self._api_client.list_integrations()

        if integration_info.name not in integrations_map:
            raise InvalidIntegrationException("Not connected to db %s!" % integration_info.name)

        # Add the load operator as a terminal node.
        apply_deltas_to_dag(
            self._dag,
            deltas=[
                AddOrReplaceOperatorDelta(
                    op=Operator(
                        id=generate_uuid(),
                        name="%s Loader" % integration_info.name,
                        description="",
                        spec=OperatorSpec(
                            load=LoadSpec(
                                service=integration_info.service,
                                integration_id=integration_info.id,
                                parameters=integration_load_params,
                            )
                        ),
                        inputs=[self._artifact_id],
                    ),
                    output_artifacts=[],
                )
            ],
        )



[docs]    def add_validation_test(
        self,
        name: str,
        description: str,
        func: ValidationTestDecoratorFunction,
        blocking: bool = False,
        file_dependencies: Optional[List[str]] = None,
    ) -> bool:
        """Attaches a validation test on this artifact.

        Args:
            name:
                The name of this unit test.
            description:
                A description of what this test is checking.
            function:
                Python function to turn into unit test. It can take any number of panda Dataframes as arguments.
                The requirements.txt file in the current directory is used. If no such file exists,
                we will infer the requirements that the function needs.
            blocking:
                Whether this test failing should also fail the entire flow.
            file_dependencies:
                A list of relative paths to files that the function needs to read.
                Python classes/methods already imported within the function's file
                need not be included.

        Returns:
            A boolean representing whether the unit test passed.

        Raises:
            InvalidRequestError:
                An error occurred because of an issue with the user's code
                or sql queries, or the `sample` parameter is malformed.
            InternalServerError:
                An unexpected error occurred within the Spiral cluster.
        """

        dir_name = func.__name__ + "_spiral_validation_test"
        delete_zip_folder_and_file(dir_name)
        zip_file_path = create_zip_file(func.__wrapped__, dir_name, file_dependencies)  # type: ignore
        funcion_spec = FunctionSpec(
            type=FunctionType.FILE,
            granularity=FunctionGranularity.TABLE,
            file_path=zip_file_path,
        )

        return self._run_validation_test(
            name=name,
            description=description,
            function_spec=funcion_spec,
            blocking=blocking,
            attach_to_dag=True,
        )



[docs]    def delete_validation_test(self, name: str) -> None:
        """Removes a validation test from this artifact.

        Args:
            name:
                The name of the unittest to remove.

        Raises:
            ValidationDoesNotExistException:
                An error occurred because the requested test cannot be found.
        """
        validation_operators = self._dag.list_operators(
            filter_to=[OperatorType.VALIDATION],
            on_artifact_id=self._artifact_id,
        )

        validation_id_by_name = {op.name: op.id for op in validation_operators}
        if name not in validation_id_by_name.keys():
            raise ValidationDoesNotExistException("Unit Test %s does not exist." % name)

        self._dag.remove_operator(
            operator_id=validation_id_by_name[name],
            must_be_type=OperatorType.VALIDATION,
        )



[docs]    def describe(self) -> None:
        """Prints out a human-readable description of the table artifact."""
        print("==================== ARTIFACT =============================")

        input_operator = self._dag.must_get_operator(with_output_artifact_id=self._artifact_id)
        load_operators = self._dag.list_operators(
            filter_to=[OperatorType.LOAD],
            on_artifact_id=self._artifact_id,
        )
        validation_operators = self._dag.list_operators(
            filter_to=[OperatorType.VALIDATION],
            on_artifact_id=self._artifact_id,
        )

        readable_dict = {
            "Id": str(self._artifact_id),
            "Label": input_operator.name,
            "Spec": input_operator.spec.json(exclude_none=True),
            "Validations": [
                {
                    "Name": op.name,
                    "Passed": op.spec.validation.test_result.check_passed,
                }
                for op in validation_operators
                if op.spec.validation and op.spec.validation.test_result
            ],
            "Destinations": [
                str(op.spec.load.integration_id)
                for op in load_operators
                if op.spec.load is not None
            ],
        }
        print(json.dumps(readable_dict, sort_keys=False, indent=4))


    def _run_validation_test(
        self,
        name: str,
        description: str,
        function_spec: FunctionSpec,
        blocking: bool = False,
        attach_to_dag: bool = False,
    ) -> bool:
        """Runs a validation test on this artifact and optionally adds it to workflow DAG.

        Args:
            name:
                The name of this unit test.
            description:
                A description of what this test is checking.
            file_path:
                The relative file path to the zip file containing the testing code.
            blocking:
                Whether this test failing should also fail the entire flow.
            make_dag_copy:
                If False, unit test is not attached to workflow dag.
                If True, unit test is attached to workflow dag.

        Returns:
            A boolean representing whether the unit test passed.

        Raises:
            InvalidRequestError:
                An error occurred because of an issue with the user's code
                or sql queries, or the `sample` parameter is malformed.
            InternalServerError:
                An unexpected error occurred within the Spiral cluster.
        """

        level = ValidationLevel.WARNING
        if blocking:
            level = ValidationLevel.ERROR

        validation_operator_id = generate_uuid()
        validation_spec = ValidationSpec(level=level, function=function_spec)
        validation_op = Operator(
            id=validation_operator_id,
            name=name,
            description=description,
            spec=OperatorSpec(
                validation=validation_spec,
            ),
            inputs=[self._artifact_id],
        )

        dag_with_validation = apply_deltas_to_dag(
            self._dag,
            deltas=[
                AddOrReplaceOperatorDelta(
                    validation_op,
                    output_artifacts=[],
                ),
            ],
            make_copy=not attach_to_dag,
        )

        preview_resp = self._api_client.preview(
            dag=apply_deltas_to_dag(
                dag_with_validation,
                deltas=[
                    SubgraphDAGDelta(
                        operator_ids=[validation_operator_id],
                        include_validation_operators=False,
                        include_load_operators=False,
                    ),
                ],
                # When pruning to the subgraph, we want to always make a copy.
                make_copy=True,
            ),
        )
        operator_result = preview_resp.operator_results[validation_operator_id]
        if operator_result.test_result is None:
            raise SpiralError("Test result not set properly.")

        if attach_to_dag:
            # Update the validation test result
            validation_spec.test_result = operator_result.test_result
            self._dag.set_validation_spec(
                operator_id=validation_operator_id,
                spec=validation_spec,
            )
        return operator_result.test_result.check_passed
